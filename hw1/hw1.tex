\section{Goal}

\begin{itemize}
	\item Install open-source DBMS on your PC. Postgres(recommend), my-sql, etc.
	\item Create at least 5 tables.
	\item Add at least 5 rows to each table.
	\item Write 10 meaningful SQL queries.
	\item Create 2 views.
\end{itemize}

\section{Example}
\Large \textbf{Tables} \normalsize \\
create table Sailors (sid int, sname varchar(20), rating int, age int); \\
create table Boats (bid int, bname varchar(20)); \\
create table Reserves (sid int, bid int, day date); \\
\\
\Large \textbf{Rows} \normalsize \\
insert into Sailors values (22, `dustin', 7, 45); \\
insert into Sailors values (28, `yuppy', 9, 35); \\
insert into Sailors values (31, `lubber', 8, 55); \\
insert into Sailors values (44, `dustin', 5, 35); \\
insert into Sailors values (58, `rusty', 10, 35); \\
\\
insert into Boats values (101, `Green'); \\
insert into Boats values (103, `Yellow'); \\
\\
insert into Reserves values (22, 101, timestamp `1996-10-10'); \\
insert into Reserves values (58, 103, timestamp `1996-11-12'); \\
\\
\Large \textbf{Meaningful queries} \normalsize \\
Find names of sailors who’ve reserved boat \#103: \\
SELECT S.sname  \\
FROM Sailors as S, Reserves as R \\
WHERE S.sid = R.sid \\
and R.bid = 103; \\
